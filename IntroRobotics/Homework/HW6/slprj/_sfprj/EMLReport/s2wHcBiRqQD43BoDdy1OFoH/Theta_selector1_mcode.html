1function [Theta_ref,Rho_ref,error] = Theta_selector(Local_max,Theta,Rho,count)
 2[H,W] = size(Local_max)

 4%mem = 100;

 7if count > 1

32

35end
36if (Theta_out > 0 && Theta_ref > 0) | (Theta_out < 0 && Theta_ref < 0)
37        error = 1;
38end
39else
40    Theta_ref=Theta(Local_max(1,1));
41    Rho_ref = Rho(Local_max(1,2));
42    Theta_out=Theta(Local_max(1,1));
